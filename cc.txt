-- [[ MENU CHÍNH SILENT ASSASSIN ]]
function OpenMainHub()
    local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()
    local Window = Rayfield:CreateWindow({Name = "TLONG HUB | SILENT ASSASSIN"})
    local Tab = Window:CreateTab("Scripts", 4483362458)
    
    local scripts = {
        ["KHA HUB"] = "https://rawscripts.net/raw/Silent-Strike-Cool-Script-Invisible-Slayer-Rounds-81109",
        ["LEE HUB"] = "https://raw.githubusercontent.com/Harrry999/slientassin/refs/heads/main/slientassinbyLee.lua",
        ["Wai Hub"] = "https://pastefy.app/MANSeU9k/raw?part=satthuthamlang.lua",
        ["Zeus Hub"] = "https://raw.githubusercontent.com/adamzeus28-coder/khangtra/refs/heads/main/script.lua"
    }

    for name, url in pairs(scripts) do
        Tab:CreateButton({Name = name, Callback = function() 
            if name == "Zeus Hub" then setclipboard("Zeus_9p1biim16u") end
            loadstring(game:HttpGet(url))() 
        end})
    end
end

-- [[ LOGIC ]]
GetKeyBtn.MouseButton1Click:Connect(function()
    -- Gọi API /getlink từ server Node.js của bạn
    local success, shortLink = pcall(function()
        return game:HttpGet(REPLIT_URL .. "/getlink?hwid=" .. playerHWID)
    end)
    setclipboard(success and shortLink or (REPLIT_URL .. "/?hwid=" .. playerHWID))
    game:GetService("StarterGui"):SetCore("SendNotification", {Title = "TLong Hub", Text = "Đã copy link vượt mã!"})
end)

KeyInput.FocusLost:Connect(function(enterPressed)
    if enterPressed then
        local success, result = pcall(function()
            return game:HttpGet(REPLIT_URL .. "/verify?key=" .. KeyInput.Text .. "&hwid=" .. playerHWID)
        end)
        if success and result == "true" then
            ScreenGui:Destroy()
            OpenMainHub() -- Mở Menu Silent Assassin
        end
    end
end)
